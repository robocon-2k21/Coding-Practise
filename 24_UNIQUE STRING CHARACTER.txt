#include <stdio.h>
#include <string.h>

void main(){
  char str[10];
  int l,n=0;
  printf("give string");
  scanf("%s",&str);
  l=strlen(str);
  for(int i=0;i<l;i++)
  {

    for(int j=0;j<l;j++)
    {
      if(str[i]==str[j])
      {
        n++;
        continue;
      }
    }
    if(n==1)
    {
      printf("%c is unique character at %d index",str[i],i);
      break;
    }                    
    n=0;
  }
  if(n==0)
  printf("unique character not found");
}
