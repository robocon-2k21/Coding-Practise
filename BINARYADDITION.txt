#include <stdio.h>
#include <conio.h>
#include <math.h>

int s=0,t=0,j=1;
int a,b,c,l,i,remainder,bin;

void main()
{
  clrscr();
  printf("Enter the first binary number:\n");
  scanf("%d",&a);
  printf("Enter the second binary number:\n");
  scanf("%d",&b);
  for(i=0;i<=4;i++)
 {
   s+=pow(2,i)*(a % 10);
   a/=10;
 }
  for(l=0;l<=4;l++)
 {
   t+=pow(2,l)*(b % 10);
   b/=10;
 }
  c=s+t;
  while(c!=0)
  {
    remainder=c % 2;
    c=c/2;
    bin=bin+remainder*j;
    j=j*10;
  }
  printf("resulted binary number is :%d\n",bin);
  getch();
}
