#include <stdio.h>

int main()
{
    int arr[]={5,5,6,8,7};
    int sum=9;
    int n=5;
    int flag=0;
    for(int i=0;i<n-1;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(arr[i]+arr[j]==sum)
            {
                flag=1;
            }
        }
    }if(flag==0)
    {
        printf("0");
    }else {
        printf("1");
    }
    

    return 0;
}
