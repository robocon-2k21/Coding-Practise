#include <stdio.h>

int main()
{
    int target;
    int arr[]={10,5,17,8,7,3,8};
    printf("enter target sum value\n");
    scanf("%d",&target);
    int n=7;
    int count=0;
    for(int i=0;i<n-1;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(arr[i]+arr[j]==target)
            {
                count=1;
            }
        }
    }
    if(count==0)
    {
        printf("0");
    }
    else {
        printf("1");
    }
    

    return 0;
}
